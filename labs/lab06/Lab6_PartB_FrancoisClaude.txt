```
CREATE VIEW orders_by_city AS 
SELECT 
  ship_city, 
  COUNT(*) 
FROM 
  orders 
GROUP BY 
  ship_city;
CREATE VIEW max_city AS 
SELECT 
  MAX(count) 
FROM 
  orders_by_city;
SELECT 
  ship_city 
FROM 
  orders_by_city 
  INNER JOIN max_city ON orders_by_city.count = max_city.max;
```

---------------------------------------- Question 2 Query ---------------------------------------- 
```
CREATE VIEW graz_orders AS 
SELECT 
  * 
FROM 
  orders 
WHERE 
  ship_city = 'Graz';
CREATE VIEW graz_details AS 
SELECT 
  order_details.order_id, 
  product_id, 
  quantity 
FROM 
  graz_orders 
  INNER JOIN order_details ON graz_orders.order_id = order_details.order_id;
CREATE VIEW min_quantity AS 
SELECT 
  MIN(quantity) 
FROM 
  graz_details;
CREATE VIEW min_product AS 
SELECT 
  * 
FROM 
  graz_details 
  INNER JOIN min_quantity ON graz_details.quantity = min_quantity.min;
SELECT 
  product_name 
FROM 
  min_product 
  INNER JOIN products on min_product.product_id = products.product_id;
```

---------------------------------------- Question 3 Query ---------------------------------------- 
```
CREATE VIEW num_orders AS 
SELECT 
  customer_id, 
  COUNT(*) 
FROM 
  orders 
GROUP BY 
  customer_id;
CREATE VIEW average_orders AS 
SELECT 
  AVG(count) 
FROM 
  num_orders;
CREATE VIEW above_avg_customer_ids AS 
SELECT 
  customer_id 
FROM 
  num_orders 
  INNER JOIN average_orders ON num_orders.count > average_orders.avg;
SELECT 
  company_name 
FROM 
  customers 
  INNER JOIN above_avg_customer_ids ON customers.customer_id = above_avg_customer_ids.customer_id;
```

```
  category_id, 
  COUNT(*) 
FROM 
  products

  category_id;

  categories.category_name,

FROM 
  categories
```
